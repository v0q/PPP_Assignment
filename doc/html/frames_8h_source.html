00005  General purpose frames per second counter for OpenGL/GLUT GNU/Linux
00006  programs. Displays "Frames per second: N" at an arbitrary position in
00007  the window. Saves and restores the app's modelview and projection matrices,
00008  colour, and lighting.

00013

00017  this file alongside your app as "frames.h", and add:
00018

00071 void frameEnd(void *font, GLclampf r, GLclampf g, GLclampf b,
00072               GLfloat x, GLfloat y, int _s) {
00073   /* font: font to use, e.g., GLUT_BITMAP_HELVETICA_10
00074      r, g, b: text colour
00075      x, y: text position in window: range [0,0] (bottom left of window)
00076            to [1,1] (top right of window). */
00077 
00078   float elapsedTime;
00079   char str[30];
00080   char *ch;
00081   GLint matrixMode;
00082   GLboolean lightingOn;
00083 
00084   gettimeofday(&frameEndTime, NULL);
00085 
00086   elapsedTime = frameEndTime.tv_sec - frameStartTime.tv_sec +
00087              ((frameEndTime.tv_usec - frameStartTime.tv_usec)/1.0E6);
00088   if(elapsedTime < 1.0/MAX_FPS)
00089   {
00090     SDL_Delay((1.0/MAX_FPS - elapsedTime) * 1000);
00091     gettimeofday(&frameEndTime, NULL);
00092 
00093     elapsedTime = frameEndTime.tv_sec - frameStartTime.tv_sec +
00094                ((frameEndTime.tv_usec - frameStartTime.tv_usec)/1.0E6);
00095   }
00096   sprintf(str, "Score: %d - %2.01ffps", _s, 1.0/elapsedTime);
00097 
00098  lightingOn = glIsEnabled(GL_LIGHTING);        /* lighting on? */
00099  if (lightingOn) glDisable(GL_LIGHTING);
00100 
00101  glGetIntegerv(GL_MATRIX_MODE, &matrixMode);  /* matrix mode? */
00102 
00103  glMatrixMode(GL_PROJECTION);
00104  glPushMatrix();
00105    glLoadIdentity();
00106    gluOrtho2D(0.0, 1.0, 0.0, 1.0);
00107    glMatrixMode(GL_MODELVIEW);
00108    glPushMatrix();
00109      glLoadIdentity();
00110      glPushAttrib(GL_COLOR_BUFFER_BIT);       /* save current colour */
00111        glColor3f(r, g, b);
00112        glRasterPos3f(x, y, 0.0);
00113        for(ch = str; *ch; ch++)
00114        {
00115           glutBitmapCharacter(font, (int)*ch);
00116        }
00117      glPopAttrib();
00118    glPopMatrix();
00119    glMatrixMode(GL_PROJECTION);
00120  glPopMatrix();
00121  glMatrixMode(matrixMode);
00122  if (lightingOn) glEnable(GL_LIGHTING);
00123 }
